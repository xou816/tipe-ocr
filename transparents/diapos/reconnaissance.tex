	\begin{equation*}
		\begin{pmatrix}
			1 \\ 0 \\ 0 \\ \vdots \\ 0
		\end{pmatrix}
		\mapsto \text{"A"}
		\qquad
		\begin{pmatrix}
			0 \\ 1 \\ 0 \\ \vdots \\ 0
		\end{pmatrix}
		\mapsto \text{"B"}.
	\end{equation*}
